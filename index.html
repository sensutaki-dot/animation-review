// server.js // Node.js express service to build a concatenated "scene" video from latest shots in Supabase Storage. // Requires ffmpeg installed in the environment and a SUPABASE_SERVICE_ROLE_KEY with storage permissions. const express = require("express"); const fs = require("fs"); const path = require("path"); const axios = require("axios"); const { spawn } = require("child_process"); const { createClient } = require("@supabase/supabase-js"); const crypto = require("crypto"); const app = express(); app.use(express.json({ limit: "10mb" })); const SUPABASE_URL = process.env.SUPABASE_URL; const SUPABASE_SERVICE_ROLE = process.env.SUPABASE_SERVICE_ROLE; // **service_role** key (keep private) const BUCKET = process.env.SUPABASE_BUCKET || "videos"; const PORT = process.env.PORT || 4000; if (!SUPABASE_URL || !SUPABASE_SERVICE_ROLE) { console.error("Faltam variáveis de ambiente: SUPABASE_URL e SUPABASE_SERVICE_ROLE"); process.exit(1); } const supabase = createClient(SUPABASE_URL, SUPABASE_SERVICE_ROLE, { global: { headers: { "x-my-app": "scene-exporter" } } }); // Utility: create a random temp dir function mkTempDir() { const dir = path.join(__dirname, "tmp", crypto.randomBytes(6).toString("hex")); fs.mkdirSync(dir, { recursive: true }); return dir; } // Determine latest versions similar to client: list root, iterate folders/files, pick highest v### async function getLatestShotsMap() { const { data: rootList, error } = await supabase.storage.from(BUCKET).list("", { limit: 2000 }); if (error) throw error; const shotsMap = {}; for (const entry of rootList) { const name = entry.name; if (name.match(/\.(mov|mp4|m4v)$/i)) { processFilePath(name, shotsMap); } else { const { data: innerList } = await supabase.storage.from(BUCKET).list(name, { limit: 2000 }); if (!innerList) { processFilePath(name, shotsMap); } else { for (const f of innerList) processFilePath(`${name}/${f.name}`, shotsMap); } } } return shotsMap; function processFilePath(p, map) { const filename = p.split("/").pop(); const shotMatch = filename.match(/(sh[_-]?\d+)/i) || (p.match(/^(sh[_-]?\d+)/i)); const versionMatch = filename.match(/_v0*(\d+)/i) || filename.match(/v0*(\d+)/i); if (!shotMatch || !versionMatch) return; const shotRaw = shotMatch[1].toUpperCase().replace(/[_-]/g, ""); const version = parseInt(versionMatch[1], 10); if (!map[shotRaw] || map[shotRaw].version < version) { map[shotRaw] = { version, path: p, fileName: filename }; } } } // Download publicly accessible file via public URL async function downloadFilePublicUrl(publicUrl, dest) { const writer = fs.createWriteStream(dest); const r = await axios.get(publicUrl, { responseType: "stream" }); r.data.pipe(writer); return new Promise((resolve, reject) => { writer.on("finish", resolve); writer.on("error", reject); }); } // Run ffmpeg command and return promise function runFFmpeg(args, cwd) { return new Promise((resolve, reject) => { const ff = spawn("ffmpeg", args, { cwd, stdio: ["ignore", "pipe", "pipe"] }); let out = "", err = ""; ff.stdout.on("data", d => out += d.toString()); ff.stderr.on("data", d => err += d.toString()); ff.on("close", code => { if (code === 0) resolve({ out, err }); else reject(new Error("ffmpeg exit " + code + "\n" + err)); }); }); } app.post("/export-scene", async (req, res) => { try { // Optional: allow client to pass shot order. If none, server builds from latest shots. const shotsOrder = req.body && Array.isArray(req.body.shots) ? req.body.shots : null; const shotsMap = await getLatestShotsMap(); const keys = shotsOrder && shotsOrder.length ? shotsOrder : Object.keys(shotsMap).sort((a,b) => { const na = parseInt(a.replace(/\D/g,''),10); const nb = parseInt(b.replace(/\D/g,''),10); return na - nb; }); if (!keys || keys.length === 0) return res.status(400).json({ error: "No shots found to export." }); const tmp = mkTempDir(); const listFile = path.join(tmp, "list.txt"); let fileEntries = []; // download each file into tmp for (let k of keys) { const info = shotsMap[k]; if (!info) continue; const parts = info.path.split("/").map(encodeURIComponent).join("/"); const publicUrl = `${SUPABASE_URL}/storage/v1/object/public/${BUCKET}/${parts}`; const destName = path.join(tmp, info.fileName.replace(/\s/g,"_")); await downloadFilePublicUrl(publicUrl, destName); // write entry for concat fileEntries.push(destName); } if (fileEntries.length === 0) return res.status(400).json({ error: "No downloadable files found." }); // create list.txt in ffmpeg concat format const listTxt = fileEntries.map(p => `file '${p.replace(/'/g,"'\\''")}'`).join("\n"); fs.writeFileSync(listFile, listTxt); const outputName = `scene_${new Date().toISOString().replace(/[:-]/g,"").replace(/\.\d+Z/,"")}.mp4`; const outputPath = path.join(tmp, outputName); // Try fast concat (works if codec/container matches) try { await runFFmpeg(["-y", "-f", "concat", "-safe", "0", "-i", listFile, "-c", "copy", outputPath], tmp); } catch (e) { // fallback: re-encode to ensure compatibility console.warn("fast concat failed, re-encoding:", e.message || e); await runFFmpeg(["-y", "-f", "concat", "-safe", "0", "-i", listFile, "-c:v", "libx264", "-preset", "veryfast", "-crf", "18", "-c:a", "aac", "-b:a", "128k", outputPath], tmp); } // upload final file back to Supabase storage (exports/) const remotePath = `exports/${outputName}`; const fileStream = fs.createReadStream(outputPath); const { data, error } = await supabase.storage.from(BUCKET).upload(remotePath, fileStream, { upsert: true }); if (error) throw error; const publicUrl = `${SUPABASE_URL}/storage/v1/object/public/${BUCKET}/${encodeURIComponent(remotePath)}`; // cleanup tmp (non-blocking) setTimeout(() => { try { fs.rmSync(tmp, { recursive: true, force: true }); } catch(e){} }, 20000); return res.json({ ok: true, url: publicUrl, path: remotePath }); } catch (err) { console.error("export error:", err); res.status(500).json({ error: err.message || String(err) }); } }); app.listen(PORT, ()=> console.log("Export service listening on", PORT));
